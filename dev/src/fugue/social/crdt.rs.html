1//! Lightweight OR-Set CRDT for collaborative playlists.
2//!
3//! Each playlist is an Observed-Remove Set where:
4//! - Elements are (track_id, owner_node, metadata)
5//! - Each add/remove carries a unique tag (node_id + lamport timestamp)
6//! - Merge: union of all adds, minus all removes. Add wins over concurrent remove.
7//!
8//! # Operation Log
9//!
10//! Every change to a collaborative playlist is stored as an operation in `crdt_ops`:
11//! - **AddTrack** — adds a track with metadata
12//! - **RemoveTrack** — removes a track
13//! - **SetName** — renames the playlist
14//!
15//! Each operation has a unique `op_id` (`{node_id}:{lamport_timestamp}`) so
16//! duplicates are safely ignored.
17//!
18//! # Materialized View
19//!
20//! After operations are stored, [`rebuild_playlist`] replays all ops in timestamp
21//! order to compute the current track list. This materialized view is what
22//! `getPlaylist` serves to clients.
23//!
24//! # Sync Protocol
25//!
26//! - When any change happens, the new CRDT ops are broadcast via gossip
27//! - When a peer connects (`NeighborUp`), ALL ops for all playlists are broadcast
28//! - [`merge_ops`] is idempotent — the same op arriving twice is ignored
29//!   (`INSERT OR IGNORE` by `op_id`)
30//! - After merging, the materialized view is rebuilt
31//!
32//! # Offline Resilience
33//!
34//! ```text
35//! Node A adds track X (offline)     Node B adds track Y (offline)
36//!   op: A:1 AddTrack(X)               op: B:1 AddTrack(Y)
37//!                   \                 /
38//!                    --- reconnect ---
39//!                   /                 \
40//!   receives B:1, merges              receives A:1, merges
41//!   rebuild: [X, Y]                   rebuild: [X, Y]
42//!   ✓ both converge                   ✓ both converge
43//! ```
44//!
45//! Both nodes end up with the same playlist — no data loss, no conflicts.

54/// A CRDT operation on a collaborative playlist.
55#[derive(Debug, Clone, Serialize, Deserialize, PartialEq)]
56pub struct CrdtOp {
57    /// Unique operation ID: "{node_id}:{lamport_clock}"
58    pub op_id: String,
59    /// Lamport timestamp for causal ordering
60    pub timestamp: u64,
61    /// The node that created this operation
62    pub origin_node: String,
63    /// The operation itself
64    pub kind: CrdtOpKind,
65}
66
67#[derive(Debug, Clone, Serialize, Deserialize, PartialEq)]
68#[serde(tag = "kind")]
69pub enum CrdtOpKind {
70    AddTrack { track: CollabTrack },
71    RemoveTrack { track_id: String, owner_node: String },
72    SetName { name: String },
73}

75/// Get the current Lamport clock for this node on a playlist, then increment it.
76pub async fn next_timestamp(
77    db: &SqlitePool,
78    playlist_id: &str,
79    _node_id: &str,
80) -> Result<u64, FugueError> {
81    // Get max timestamp we've seen for this playlist
82    let row: Option<(i64,)> = sqlx::query_as(
83        "SELECT MAX(timestamp) FROM crdt_ops WHERE playlist_id = ?",
84    )
85    .bind(playlist_id)
86    .fetch_optional(db)
87    .await?;
88
89    let current = row.and_then(|(t,)| Some(t)).unwrap_or(0) as u64;
90    Ok(current + 1)
91}

93/// Store an operation (idempotent — ignores duplicates by op_id).
94pub async fn store_op(
95    db: &SqlitePool,
96    playlist_id: &str,
97    op: &CrdtOp,
98) -> Result<bool, FugueError> {
99    let op_json = serde_json::to_string(&op.kind)
100        .map_err(|e| FugueError::Internal(format!("serialize op: {e}")))?;
101
102    let result = sqlx::query(
103        "INSERT OR IGNORE INTO crdt_ops (playlist_id, op_id, timestamp, origin_node, op_json)
104         VALUES (?, ?, ?, ?, ?)",
105    )
106    .bind(playlist_id)
107    .bind(&op.op_id)
108    .bind(op.timestamp as i64)
109    .bind(&op.origin_node)
110    .bind(&op_json)
111    .execute(db)
112    .await?;
113
114    Ok(result.rows_affected() > 0)
115}

117/// Get all operations for a playlist (for sync).
118pub async fn get_all_ops(
119    db: &SqlitePool,
120    playlist_id: &str,
121) -> Result<Vec<CrdtOp>, FugueError> {
122    let rows: Vec<(String, i64, String, String)> = sqlx::query_as(
123        "SELECT op_id, timestamp, origin_node, op_json FROM crdt_ops
124         WHERE playlist_id = ? ORDER BY timestamp ASC",
125    )
126    .bind(playlist_id)
127    .fetch_all(db)
128    .await?;
129
130    let ops: Vec<CrdtOp> = rows
131        .into_iter()
132        .filter_map(|(op_id, timestamp, origin_node, op_json)| {
133            let kind: CrdtOpKind = serde_json::from_str(&op_json).ok()?;
134            Some(CrdtOp {
135                op_id,
136                timestamp: timestamp as u64,
137                origin_node,
138                kind,
139            })
140        })
141        .collect();
142
143    Ok(ops)
144}

146/// Merge a set of remote operations into our local state.
147/// Returns the number of new operations applied.
148pub async fn merge_ops(
149    db: &SqlitePool,
150    playlist_id: &str,
151    ops: &[CrdtOp],
152) -> Result<usize, FugueError> {
153    let mut new_count = 0;
154    for op in ops {
155        if store_op(db, playlist_id, op).await? {
156            new_count += 1;
157        }
158    }
159
160    if new_count > 0 {
161        // Rebuild the materialized playlist from the op log
162        rebuild_playlist(db, playlist_id).await?;
163        debug!("crdt: merged {} new ops for playlist {}", new_count, playlist_id);
164    }
165
166    Ok(new_count)
167}

169/// Rebuild the materialized playlist tracks from the CRDT operation log.
170/// This is the "resolve" step: replay all ops to compute current state.
171pub async fn rebuild_playlist(
172    db: &SqlitePool,
173    playlist_id: &str,
174) -> Result<(), FugueError> {
175    let ops = get_all_ops(db, playlist_id).await?;
176
177    // Compute the current state by replaying ops
178    let mut tracks: Vec<CollabTrack> = Vec::new();
179    let mut removed: std::collections::HashSet<(String, String)> = std::collections::HashSet::new();
180    let mut playlist_name: Option<String> = None;
181
182    for op in &ops {
183        match &op.kind {
184            CrdtOpKind::AddTrack { track } => {
185                let key = (track.track_id.clone(), track.owner_node.clone());
186                if !removed.contains(&key) {
187                    // Remove existing entry with same key (re-add updates metadata)
188                    tracks.retain(|t| !(t.track_id == track.track_id && t.owner_node == track.owner_node));
189                    tracks.push(track.clone());
190                }
191            }
192            CrdtOpKind::RemoveTrack { track_id, owner_node } => {
193                let key = (track_id.clone(), owner_node.clone());
194                removed.insert(key);
195                tracks.retain(|t| !(t.track_id == *track_id && t.owner_node == *owner_node));
196            }
197            CrdtOpKind::SetName { name } => {
198                playlist_name = Some(name.clone());
199            }
200        }
201    }
202
203    // Update the materialized tables
204    sqlx::query("DELETE FROM collab_playlist_tracks WHERE playlist_id = ?")
205        .bind(playlist_id)
206        .execute(db)
207        .await?;
208
209    for (pos, track) in tracks.iter().enumerate() {
210        sqlx::query(
211            "INSERT OR IGNORE INTO collab_playlist_tracks
212             (playlist_id, track_id, owner_node, title, artist, album, duration, position, added_by)
213             VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)",
214        )
215        .bind(playlist_id)
216        .bind(&track.track_id)
217        .bind(&track.owner_node)
218        .bind(&track.title)
219        .bind(&track.artist)
220        .bind(&track.album)
221        .bind(track.duration)
222        .bind(pos as i64)
223        .bind(&track.added_by)
224        .execute(db)
225        .await?;
226    }
227
228    if let Some(name) = playlist_name {
229        sqlx::query("UPDATE collab_playlists SET name = ?, updated_at = datetime('now') WHERE id = ?")
230            .bind(&name)
231            .bind(playlist_id)
232            .execute(db)
233            .await?;
234    }
235
236    debug!("crdt: rebuilt playlist {} with {} tracks", playlist_id, tracks.len());
237    Ok(())
238}
