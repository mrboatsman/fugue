9fn to_iso8601(ts: &str) -> String {
10    if ts.contains('T') { return ts.to_string(); }
11    ts.replace(' ', "T") + "Z"
12}

14/// A track in a collaborative playlist, with ownership info.
15#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
16pub struct CollabTrack {
17    pub track_id: String,      // namespaced ID on the owner's node
18    pub owner_node: String,    // node_id that can stream this track
19    pub title: String,
20    pub artist: Option<String>,
21    pub album: Option<String>,
22    pub duration: Option<i64>,
23    pub added_by: String,      // node_id that added this
24}

26/// Gossip operations for collaborative playlists.
27#[derive(Debug, Clone, Serialize, Deserialize)]
28#[serde(tag = "op")]
29pub enum PlaylistOp {
30    /// Create a new collaborative playlist.
31    Create {
32        playlist_id: String,
33        name: String,
34    },
35    /// Add a track to the playlist.
36    AddTrack {
37        playlist_id: String,
38        track: CollabTrack,
39    },
40    /// Remove a track from the playlist by position.
41    RemoveTrack {
42        playlist_id: String,
43        track_id: String,
44        owner_node: String,
45    },
46    /// Rename the playlist.
47    Rename {
48        playlist_id: String,
49        name: String,
50    },
51    /// Delete the playlist.
52    Delete {
53        playlist_id: String,
54    },
55    /// Full sync: send all tracks (used when a new friend joins).
56    FullSync {
57        playlist_id: String,
58        name: String,
59        tracks: Vec<CollabTrack>,
60    },
61}

83pub async fn add_track(
84    db: &SqlitePool,
85    playlist_id: &str,
86    track: &CollabTrack,
87) -> Result<(), FugueError> {
88    // Get next position
89    let max_pos: (i64,) = sqlx::query_as(
90        "SELECT COALESCE(MAX(position), -1) FROM collab_playlist_tracks WHERE playlist_id = ?",
91    )
92    .bind(playlist_id)
93    .fetch_one(db)
94    .await?;
95
96    sqlx::query(
97        "INSERT OR IGNORE INTO collab_playlist_tracks
98         (playlist_id, track_id, owner_node, title, artist, album, duration, position, added_by)
99         VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)",
100    )
101    .bind(playlist_id)
102    .bind(&track.track_id)
103    .bind(&track.owner_node)
104    .bind(&track.title)
105    .bind(&track.artist)
106    .bind(&track.album)
107    .bind(track.duration)
108    .bind(max_pos.0 + 1)
109    .bind(&track.added_by)
110    .execute(db)
111    .await?;
112
113    // Update playlist timestamp
114    sqlx::query("UPDATE collab_playlists SET updated_at = datetime('now') WHERE id = ?")
115        .bind(playlist_id)
116        .execute(db)
117        .await?;
118
119    debug!("collab: added track '{}' to playlist {}", track.title, playlist_id);
120    Ok(())
121}

123pub async fn remove_track(
124    db: &SqlitePool,
125    playlist_id: &str,
126    track_id: &str,
127    owner_node: &str,
128) -> Result<(), FugueError> {
129    sqlx::query(
130        "DELETE FROM collab_playlist_tracks WHERE playlist_id = ? AND track_id = ? AND owner_node = ?",
131    )
132    .bind(playlist_id)
133    .bind(track_id)
134    .bind(owner_node)
135    .execute(db)
136    .await?;
137
138    // Reindex positions
139    let tracks: Vec<(i64,)> = sqlx::query_as(
140        "SELECT rowid FROM collab_playlist_tracks WHERE playlist_id = ? ORDER BY position",
141    )
142    .bind(playlist_id)
143    .fetch_all(db)
144    .await?;
145
146    for (i, (rowid,)) in tracks.iter().enumerate() {
147        sqlx::query("UPDATE collab_playlist_tracks SET position = ? WHERE rowid = ?")
148            .bind(i as i64)
149            .bind(rowid)
150            .execute(db)
151            .await?;
152    }
153
154    Ok(())
155}

183/// Get all collaborative playlists. Sets `owner` to `username` for playlists
184/// the user can edit, so Subsonic clients allow adding songs.
185pub async fn list_playlists(db: &SqlitePool, username: &str, node_id: &str) -> Result<Vec<serde_json::Value>, FugueError> {
186    let rows: Vec<(String, String, String, String)> = sqlx::query_as(
187        "SELECT id, name, created_by, created_at FROM collab_playlists ORDER BY name",
188    )
189    .fetch_all(db)
190    .await?;
191
192    let mut playlists = Vec::new();
193    for (id, name, created_by, created_at) in rows {
194        let track_count: (i64,) = sqlx::query_as(
195            "SELECT COUNT(*) FROM collab_playlist_tracks WHERE playlist_id = ?",
196        )
197        .bind(&id)
198        .fetch_one(db)
199        .await?;
200
201        let duration: (Option<i64>,) = sqlx::query_as(
202            "SELECT SUM(duration) FROM collab_playlist_tracks WHERE playlist_id = ?",
203        )
204        .bind(&id)
205        .fetch_one(db)
206        .await?;
207
208        // Encode as collab playlist ID: "c:{uuid}"
209        let encoded_id = encode_collab_id(&id);
210
211        // Check if the current user can edit this playlist.
212        // If so, set owner to their username so clients allow editing.
213        let can_edit = can_edit(db, &id, node_id).await.unwrap_or(false)
214            || created_by == node_id;
215        let display_owner = if can_edit { username } else { &created_by };
216
217        playlists.push(serde_json::json!({
218            "id": encoded_id,
219            "name": format!("[Collab] {}", name),
220            "comment": format!("Collaborative playlist (created by {})", created_by),
221            "public": true,
222            "owner": display_owner,
223            "songCount": track_count.0,
224            "duration": duration.0.unwrap_or(0),
225            "created": to_iso8601(&created_at),
226            "changed": to_iso8601(&created_at),
227        }));
228    }
229
230    Ok(playlists)
231}

233/// Get a specific collaborative playlist with all tracks.
234pub async fn get_playlist(
235    db: &SqlitePool,
236    playlist_id: &str,
237    username: &str,
238    node_id: &str,
239) -> Result<Option<serde_json::Value>, FugueError> {
240    let row: Option<(String, String, String, String)> = sqlx::query_as(
241        "SELECT id, name, created_by, created_at FROM collab_playlists WHERE id = ?",
242    )
243    .bind(playlist_id)
244    .fetch_optional(db)
245    .await?;
246
247    let (id, name, created_by, created_at) = match row {
248        Some(r) => r,
249        None => return Ok(None),
250    };
251
252    let tracks: Vec<(String, String, String, Option<String>, Option<String>, Option<i64>, i64, String)> =
253        sqlx::query_as(
254            "SELECT track_id, owner_node, title, artist, album, duration, position, added_by
255             FROM collab_playlist_tracks WHERE playlist_id = ? ORDER BY position",
256        )
257        .bind(playlist_id)
258        .fetch_all(db)
259        .await?;
260
261    let entries: Vec<serde_json::Value> = tracks
262        .iter()
263        .map(|(track_id, owner_node, title, artist, album, duration, _pos, _added_by)| {
264            // Encode the track ID with owner info so we can route streaming
265            let stream_id = encode_remote_track_id(owner_node, track_id);
266            serde_json::json!({
267                "id": stream_id,
268                "title": title,
269                "artist": artist.as_deref().unwrap_or(""),
270                "album": album.as_deref().unwrap_or(""),
271                "duration": duration.unwrap_or(0),
272                "isDir": false,
273                "coverArt": stream_id,
274            })
275        })
276        .collect();
277
278    let encoded_id = encode_collab_id(&id);
279    let user_can_edit = can_edit(db, playlist_id, node_id).await.unwrap_or(false)
280        || created_by == node_id;
281    let display_owner = if user_can_edit { username } else { &created_by };
282
283    Ok(Some(serde_json::json!({
284        "playlist": {
285            "id": encoded_id,
286            "name": format!("[Collab] {}", name),
287            "comment": format!("Collaborative playlist (created by {})", created_by),
288            "public": true,
289            "owner": display_owner,
290            "songCount": entries.len(),
291            "duration": tracks.iter().filter_map(|t| t.5).sum::<i64>(),
292            "created": to_iso8601(&created_at),
293            "changed": to_iso8601(&created_at),
294            "entry": entries,
295        }
296    })))
297}

329use base64::Engine;
330use base64::engine::general_purpose::URL_SAFE_NO_PAD;

332/// Encode a collaborative playlist ID: base64("c:{uuid}")
333pub fn encode_collab_id(uuid: &str) -> String {
334    URL_SAFE_NO_PAD.encode(format!("c:{uuid}").as_bytes())
335}
336
337/// Decode a collaborative playlist ID.
338pub fn decode_collab_id(encoded: &str) -> Option<String> {
339    let bytes = URL_SAFE_NO_PAD.decode(encoded).ok()?;
340    let raw = String::from_utf8(bytes).ok()?;
341    raw.strip_prefix("c:").map(|s| s.to_string())
342}
343
344/// Encode a remote track ID: base64("r:{owner_node}:{track_id}")
345/// This tells the streaming layer to fetch via Iroh from the owner node.
346pub fn encode_remote_track_id(owner_node: &str, track_id: &str) -> String {
347    URL_SAFE_NO_PAD.encode(format!("r:{owner_node}:{track_id}").as_bytes())
348}
349
350/// Decode a remote track ID. Returns (owner_node, track_id).
351pub fn decode_remote_track_id(encoded: &str) -> Option<(String, String)> {
352    let bytes = URL_SAFE_NO_PAD.decode(encoded).ok()?;
353    let raw = String::from_utf8(bytes).ok()?;
354    let rest = raw.strip_prefix("r:")?;
355    let (owner, track) = rest.split_once(':')?;
356    Some((owner.to_string(), track.to_string()))
357}
358
359/// Check if an ID is a remote track ID.
360pub fn is_remote_track_id(encoded: &str) -> bool {
361    decode_remote_track_id(encoded).is_some()
362}

366/// Roles for collaborative playlist members.
367#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
368#[serde(rename_all = "lowercase")]
369pub enum Role {
370    Owner,
371    Collab,
372    Viewer,
373}
374
375impl Role {
376    pub fn as_str(&self) -> &str {
377        match self {
378            Role::Owner => "owner",
379            Role::Collab => "collab",
380            Role::Viewer => "viewer",
381        }
382    }
383
384    pub fn from_str(s: &str) -> Option<Self> {
385        match s {
386            "owner" => Some(Self::Owner),
387            "collab" => Some(Self::Collab),
388            "viewer" => Some(Self::Viewer),
389            _ => None,
390        }
391    }
392
393    pub fn can_edit(&self) -> bool {
394        matches!(self, Role::Owner | Role::Collab)
395    }
396}

398/// Generate an invite code for a playlist with a specific role.
399/// Format: base64("i:{playlist_id}:{role}:{name}:{sender_name}:{sender_ticket}")
400/// The sender's name and ticket are embedded so the recipient can auto-add them
401/// as a friend without a prior ticket exchange.
402pub fn generate_invite(
403    playlist_id: &str,
404    role: Role,
405    name: &str,
406    sender_name: &str,
407    sender_ticket: &str,
408) -> String {
409    URL_SAFE_NO_PAD.encode(
410        format!(
411            "i:{}:{}:{}:{}:{}",
412            playlist_id,
413            role.as_str(),
414            name,
415            sender_name,
416            sender_ticket,
417        )
418        .as_bytes(),
419    )
420}

422/// Parsed invite code data.
423pub struct InviteInfo {
424    pub playlist_id: String,
425    pub role: Role,
426    pub name: String,
427    /// Sender's display name (None for legacy codes without sender info).
428    pub sender_name: Option<String>,
429    /// Sender's Iroh ticket (None for legacy codes without sender info).
430    pub sender_ticket: Option<String>,
431}
432
433/// Parse an invite code. Backward compatible with old format (no sender info).
434pub fn parse_invite(code: &str) -> Option<InviteInfo> {
435    let bytes = URL_SAFE_NO_PAD.decode(code).ok()?;
436    let raw = String::from_utf8(bytes).ok()?;
437    let rest = raw.strip_prefix("i:")?;
438    // Split: playlist_id:role:remaining
439    let (playlist_id, remainder) = rest.split_once(':')?;
440    let (role_str, after_role) = remainder.split_once(':')?;
441    let role = Role::from_str(role_str)?;
442
443    // Try to split remaining into name:sender_name:sender_ticket
444    // The sender_ticket contains colons (it's a named ticket like "Name:base64"),
445    // so we split from the right: the last segment is sender_ticket, second-to-last
446    // is sender_name, everything before is the playlist name.
447    // But sender_ticket itself may contain ":", so we use a different approach:
448    // Count fields. The sender_ticket is base64url (no colons), but the named
449    // ticket format is "name:base64" which has exactly one colon.
450    // Strategy: try splitting into 3+ parts. If we have at least 3 parts after role,
451    // the last two are sender_name and sender_ticket.
452    let parts: Vec<&str> = after_role.splitn(4, ':').collect();
453    match parts.len() {
454        1 => {
455            // Legacy format: just name
456            Some(InviteInfo {
457                playlist_id: playlist_id.to_string(),
458                role,
459                name: parts[0].to_string(),
460                sender_name: None,
461                sender_ticket: None,
462            })
463        }
464        3.. => {
465            // New format: name:sender_name:sender_ticket (ticket may contain colons)
466            let name = parts[0].to_string();
467            let sender_name = parts[1].to_string();
468            // Rejoin remaining parts as the sender ticket (it may have colons in named format)
469            let ticket_start = after_role
470                .find(':')
471                .and_then(|p1| after_role[p1 + 1..].find(':').map(|p2| p1 + 1 + p2 + 1))
472                .unwrap_or(after_role.len());
473            let sender_ticket = after_role[ticket_start..].to_string();
474            Some(InviteInfo {
475                playlist_id: playlist_id.to_string(),
476                role,
477                name,
478                sender_name: Some(sender_name),
479                sender_ticket: if sender_ticket.is_empty() { None } else { Some(sender_ticket) },
480            })
481        }
482        _ => {
483            // 2 parts: ambiguous, treat as legacy
484            Some(InviteInfo {
485                playlist_id: playlist_id.to_string(),
486                role,
487                name: after_role.to_string(),
488                sender_name: None,
489                sender_ticket: None,
490            })
491        }
492    }
493}

495/// Generate a friend code for connecting without a playlist context.
496/// Format: base64("f:{display_name}:{ticket}")
497pub fn generate_friend_code(display_name: &str, ticket: &str) -> String {
498    URL_SAFE_NO_PAD.encode(format!("f:{}:{}", display_name, ticket).as_bytes())
499}
500
501/// Parse a friend code. Returns (display_name, ticket).
502pub fn parse_friend_code(code: &str) -> Option<(String, String)> {
503    let bytes = URL_SAFE_NO_PAD.decode(code).ok()?;
504    let raw = String::from_utf8(bytes).ok()?;
505    let rest = raw.strip_prefix("f:")?;
506    let (name, ticket) = rest.split_once(':')?;
507    if ticket.is_empty() {
508        return None;
509    }
510    Some((name.to_string(), ticket.to_string()))
511}

522        "INSERT INTO collab_playlist_members (playlist_id, node_id, name, role)

525    )

527    .bind(node_id)
528    .bind(name)

530    .execute(db)

533    Ok(())
534}
