1//! Party mode: synchronized group playback.
2//!
3//! One host controls playback; followers track the host's state with
4//! sub-second accuracy via gossip heartbeats and clock offset estimation.
5//!
6//! Sessions are ephemeral (in-memory only) — they don't survive restarts,
7//! which is appropriate for a live listening feature.
8

18fn uuid_v4() -> String {
19    use rand::Rng;
20    let mut rng = rand::rng();
21    let mut bytes = [0u8; 16];
22    rng.fill(&mut bytes);
23    bytes[6] = (bytes[6] & 0x0f) | 0x40;
24    bytes[8] = (bytes[8] & 0x3f) | 0x80;
25    format!(
26        "{:02x}{:02x}{:02x}{:02x}-{:02x}{:02x}-{:02x}{:02x}-{:02x}{:02x}-{:02x}{:02x}{:02x}{:02x}{:02x}{:02x}",
27        bytes[0], bytes[1], bytes[2], bytes[3],
28        bytes[4], bytes[5], bytes[6], bytes[7],
29        bytes[8], bytes[9], bytes[10], bytes[11],
30        bytes[12], bytes[13], bytes[14], bytes[15],
31    )
32}

75/// State for a node that is following a party session.
76#[derive(Debug)]
77pub struct FollowingSession {
78    pub session_id: String,
79    pub host_node_id: String,
80    pub host_name: String,
81    /// Running clock offset estimate: `host_time - local_time` in ms.
82    pub clock_offset_ms: i64,
83    /// Estimated round-trip time in ms.
84    pub rtt_ms: u64,
85    /// Recent NTP-style samples for weighted-median filtering.
86    clock_samples: VecDeque<ClockSample>,
87    /// Last applied sequence number.
88    pub last_seq: u64,
89}
90
91impl FollowingSession {
92    pub fn new(session_id: String, host_node_id: String, host_name: String) -> Self {
93        Self {
94            session_id,
95            host_node_id,
96            host_name,
97            clock_offset_ms: 0,
98            rtt_ms: 0,
99            clock_samples: VecDeque::with_capacity(10),
100            last_seq: 0,
101        }
102    }
103
104    /// NTP 4-timestamp clock offset estimation.
105    /// T1 = client_send_ms, T2 = server_recv_ms, T3 = server_send_ms,
106    /// T4 = client_recv_ms (local time when pong arrived).
107    ///
108    /// offset = ((T2 - T1) + (T3 - T4)) / 2   (host_time - local_time)
109    /// rtt    = (T4 - T1) - (T3 - T2)
110    pub fn add_clock_sample(&mut self, t1: u64, t2: u64, t3: u64, t4: u64) {
111        let rtt = (t4 as i64 - t1 as i64) - (t3 as i64 - t2 as i64);
112        let rtt = rtt.max(0) as u64;
113        let offset = ((t2 as i64 - t1 as i64) + (t3 as i64 - t4 as i64)) / 2;
114
115        if self.clock_samples.len() >= 10 {
116            self.clock_samples.pop_front();
117        }
118        self.clock_samples.push_back(ClockSample { offset_ms: offset, rtt_ms: rtt });
119
120        // Weighted median: prefer low-RTT samples (less jitter).
121        // Sort by offset, weight inversely by RTT.
122        self.clock_offset_ms = weighted_median(&self.clock_samples);
123        // RTT as exponential moving average
124        if self.rtt_ms == 0 {
125            self.rtt_ms = rtt;
126        } else {
127            self.rtt_ms = (self.rtt_ms * 7 + rtt * 3) / 10;
128        }
129        debug!("clock: offset={}ms rtt={}ms (sample: offset={}ms rtt={}ms)",
130            self.clock_offset_ms, self.rtt_ms, offset, rtt);
131    }
132
133    /// Legacy: update from one-way timestamp (gossip path, less accurate).
134    pub fn update_clock_offset(&mut self, host_timestamp_ms: u64) {
135        let local_now_ms = now_ms();
136        let sample = host_timestamp_ms as i64 - local_now_ms as i64;
137        // Treat one-way as a high-RTT sample (less trusted)
138        if self.clock_samples.len() >= 10 {
139            self.clock_samples.pop_front();
140        }
141        self.clock_samples.push_back(ClockSample {
142            offset_ms: sample,
143            rtt_ms: 500, // assume high RTT for one-way estimates
144        });
145        self.clock_offset_ms = weighted_median(&self.clock_samples);
146    }
147}

160/// Top-level party state for a Fugue node.
161/// A node can host at most one session and follow at most one session
162/// (but not both simultaneously).
163#[derive(Debug, Default)]
164pub struct PartyState {
165    pub hosting: Option<PartySession>,
166    pub following: Option<FollowingSession>,
167    /// Remote parties seen via gossip. Entries expire after 60s without a heartbeat.
168    pub active_parties: Vec<ActiveParty>,
169}
170
171impl PartyState {
172    /// Create a new party session hosted by this node.
173    pub fn create_session(&mut self, host_name: String, host_node_id: String) -> &PartySession {
174        let session_id = uuid_v4();
175        self.following = None; // can't follow while hosting
176        self.hosting = Some(PartySession {
177            session_id,
178            host_name,
179            host_node_id,
180            members: Vec::new(),
181            seq: 0,
182            state: PartyPlaybackState::Stopped,
183            track: None,
184            position_secs: 0.0,
185            dj_timestamp_ms: 0,
186            playlist: Vec::new(),
187            playlist_index: 0,
188            queue: Vec::new(),
189            queue_index: 0,
190            playing_from_queue: false,
191            dj_endpoint_addr: None,
192        });
193        self.hosting.as_ref().unwrap()
194    }
195
196    /// End the hosted session.
197    pub fn end_session(&mut self) -> Option<String> {
198        self.hosting.take().map(|s| s.session_id)
199    }
200
201    /// Start following a session.
202    pub fn follow(
203        &mut self,
204        session_id: String,
205        host_node_id: String,
206        host_name: String,
207    ) {
208        self.hosting = None; // can't host while following
209        self.following = Some(FollowingSession::new(session_id, host_node_id, host_name));
210    }
211
212    /// Stop following.
213    pub fn unfollow(&mut self) -> Option<String> {
214        self.following.take().map(|s| s.session_id)
215    }
216
217    /// Record or refresh a remote party seen via gossip.
218    pub fn touch_active_party(&mut self, session_id: &str, host_name: &str, host_node_id: &str) {
219        let now = now_ms();
220        if let Some(p) = self.active_parties.iter_mut().find(|p| p.session_id == session_id) {
221            p.last_seen_ms = now;
222            p.host_name = host_name.to_string();
223        } else {
224            self.active_parties.push(ActiveParty {
225                session_id: session_id.to_string(),
226                host_name: host_name.to_string(),
227                host_node_id: host_node_id.to_string(),
228                last_seen_ms: now,
229            });
230        }
231    }
232
233    /// Remove a party (when PartyEnd is received).
234    pub fn remove_active_party(&mut self, session_id: &str) {
235        self.active_parties.retain(|p| p.session_id != session_id);
236    }
237
238    /// Prune parties with no heartbeat in the last 60 seconds.
239    pub fn prune_stale_parties(&mut self) {
240        let cutoff = now_ms().saturating_sub(60_000);
241        self.active_parties.retain(|p| p.last_seen_ms > cutoff);
242    }
243
244    /// Get active parties (excluding our own if hosting).
245    pub fn discover_parties(&mut self) -> Vec<ActiveParty> {
246        self.prune_stale_parties();
247        let own_session = self.hosting.as_ref().map(|h| h.session_id.as_str());
248        self.active_parties
249            .iter()
250            .filter(|p| own_session != Some(p.session_id.as_str()))
251            .cloned()
252            .collect()
253    }
254
255    /// Add a member to the hosted session.
256    pub fn add_member(&mut self, node_id: &str, display_name: &str) {
257        if let Some(ref mut session) = self.hosting {
258            // Don't add duplicates
259            if !session.members.iter().any(|m| m.node_id == node_id) {
260                session.members.push(PartyMember {
261                    node_id: node_id.to_string(),
262                    display_name: display_name.to_string(),
263                });
264            }
265        }
266    }
267
268    /// Remove a member from the hosted session.
269    pub fn remove_member(&mut self, node_id: &str) {
270        if let Some(ref mut session) = self.hosting {
271            session.members.retain(|m| m.node_id != node_id);
272        }
273    }
274}

278/// Current time in milliseconds since UNIX epoch.
279pub fn now_ms() -> u64 {
280    SystemTime::now()
281        .duration_since(UNIX_EPOCH)
282        .unwrap_or_default()
283        .as_millis() as u64
284}

286/// Compute weighted median: lower-RTT samples are preferred (less jitter).
287/// Weight = 1 / (rtt + 1). Finds the offset at which cumulative weight >= half.
288fn weighted_median(samples: &VecDeque<ClockSample>) -> i64 {
289    if samples.is_empty() {
290        return 0;
291    }
292    let mut entries: Vec<(i64, f64)> = samples
293        .iter()
294        .map(|s| (s.offset_ms, 1.0 / (s.rtt_ms as f64 + 1.0)))
295        .collect();
296    entries.sort_unstable_by_key(|&(offset, _)| offset);
297    let total_weight: f64 = entries.iter().map(|&(_, w)| w).sum();
298    let half = total_weight / 2.0;
299    let mut cumulative = 0.0;
300    for &(offset, weight) in &entries {
301        cumulative += weight;
302        if cumulative >= half {
303            return offset;
304        }
305    }
306    entries.last().map(|&(o, _)| o).unwrap_or(0)
307}

311/// Resolve a `PartyTrack` to a local Subsonic song ID.
312///
313/// Strategy:
314/// 1. Dedup fingerprint lookup in `dedup_groups` → canonical_id
315/// 2. Metadata search in `tracks` by artist + title (case-insensitive)
316/// 3. `None` if not found (caller should skip this track)
317pub async fn resolve_track(db: &SqlitePool, track: &PartyTrack) -> Option<String> {
318    // Strategy 1: dedup fingerprint
319    if let Some(ref fp) = track.fingerprint {
320        let row: Option<(String,)> = sqlx::query_as(
321            "SELECT canonical_id FROM dedup_groups WHERE fingerprint = ?",
322        )
323        .bind(fp)
324        .fetch_optional(db)
325        .await
326        .ok()?;
327        if let Some((canonical_id,)) = row {
328            debug!("party: resolved track by fingerprint: {} → {}", fp, canonical_id);
329            return Some(canonical_id);
330        }
331    }
332
333    // Strategy 2: metadata search — try exact match on title + artist
334    let row: Option<(String,)> = sqlx::query_as(
335        "SELECT id FROM tracks WHERE LOWER(title) = LOWER(?) AND LOWER(artist) = LOWER(?) LIMIT 1",
336    )
337    .bind(&track.title)
338    .bind(&track.artist)
339    .fetch_optional(db)
340    .await
341    .ok()?;
342
343    if let Some((id,)) = row {
344        debug!(
345            "party: resolved track by metadata: {} - {} → {}",
346            track.artist, track.title, id
347        );
348        return Some(id);
349    }
350
351    debug!(
352        "party: could not resolve track: {} - {} ({})",
353        track.artist,
354        track.title,
355        track.fingerprint.as_deref().unwrap_or("no fingerprint")
356    );
357    None
358}

362/// Serializable party status for the admin endpoint.
363#[derive(Serialize)]
364pub struct PartyStatus {
365    pub mode: &'static str, // "off", "hosting", "following"
366    #[serde(skip_serializing_if = "Option::is_none")]
367    pub session_id: Option<String>,
368    #[serde(skip_serializing_if = "Option::is_none")]
369    pub host_name: Option<String>,
370    #[serde(skip_serializing_if = "Option::is_none")]
371    pub members: Option<Vec<PartyMember>>,
372    #[serde(skip_serializing_if = "Option::is_none")]
373    pub state: Option<PartyPlaybackState>,
374    #[serde(skip_serializing_if = "Option::is_none")]
375    pub track: Option<PartyTrack>,
376    #[serde(skip_serializing_if = "Option::is_none")]
377    pub position_secs: Option<f64>,
378}
379
380impl PartyStatus {
381    pub fn from_state(party: &PartyState) -> Self {
382        match (&party.hosting, &party.following) {
383            (Some(h), _) => Self {
384                mode: "hosting",
385                session_id: Some(h.session_id.clone()),
386                host_name: Some(h.host_name.clone()),
387                members: Some(h.members.clone()),
388                state: Some(h.state),
389                track: h.track.clone(),
390                position_secs: Some(h.position_secs),
391            },
392            (_, Some(f)) => Self {
393                mode: "following",
394                session_id: Some(f.session_id.clone()),
395                host_name: Some(f.host_name.clone()),
396                members: None,
397                state: None,
398                track: None,
399                position_secs: None,
400            },
401            _ => Self {
402                mode: "off",
403                session_id: None,
404                host_name: None,
405                members: None,
406                state: None,
407                track: None,
408                position_secs: None,
409            },
410        }
411    }
412}

418    #[test]
419    fn test_weighted_median() {
420        let mut samples = VecDeque::new();
421        // Empty → 0
422        assert_eq!(weighted_median(&samples), 0);
423        // Single sample
424        samples.push_back(ClockSample { offset_ms: 100, rtt_ms: 10 });
425        assert_eq!(weighted_median(&samples), 100);
426        // Low-RTT sample should be preferred
427        samples.push_back(ClockSample { offset_ms: 50, rtt_ms: 1 });
428        samples.push_back(ClockSample { offset_ms: 200, rtt_ms: 100 });
429        // 50 has weight 0.5, 100 has weight ~0.09, 200 has weight ~0.01
430        // sorted by offset: 50 (w=0.5), 100 (w=0.09), 200 (w=0.01)
431        // total ≈ 0.6, half ≈ 0.3. Cumulative passes half at offset=50
432        assert_eq!(weighted_median(&samples), 50);
433    }

435    #[test]
436    fn test_ntp_clock_offset() {
437        let mut session = FollowingSession::new(
438            "test".into(),
439            "node1".into(),
440            "Alice".into(),
441        );
442        assert_eq!(session.clock_offset_ms, 0);
443
444        // Simulate NTP: host clock is 50ms ahead, 10ms network each way
445        // T1=1000, T2=1060 (host received: 1000+10+50), T3=1061, T4=1071 (1061-50+10+50=1071)
446        // Actually let's be precise: T4 = T1 + RTT = 1000 + 20 = 1020... but host is 50 ahead
447        // T1=1000 (client sends)
448        // T2=1060 (server receives: client_time+10 in server_clock = 1000+10+50 = 1060)
449        // T3=1061 (server sends, 1ms processing)
450        // T4=1021 (client receives: T3-50+10 = 1021)
451        // offset = ((T2-T1)+(T3-T4))/2 = ((60)+(40))/2 = 50 ✓
452        // RTT = (T4-T1)-(T3-T2) = (21)-(1) = 20 ✓
453        session.add_clock_sample(1000, 1060, 1061, 1021);
454        assert_eq!(session.clock_offset_ms, 50);
455        assert_eq!(session.rtt_ms, 20);
456    }

458    #[test]
459    fn test_party_state_lifecycle() {
460        let mut state = PartyState::default();
461        assert!(state.hosting.is_none());
462        assert!(state.following.is_none());
463
464        // Create session
465        let session = state.create_session("Alice".into(), "node_a".into());
466        assert_eq!(session.host_name, "Alice");
467
468        // Add members
469        state.add_member("node_b", "Bob");
470        state.add_member("node_c", "Carol");
471        assert_eq!(state.hosting.as_ref().unwrap().members.len(), 2);
472
473        // No duplicates
474        state.add_member("node_b", "Bob");
475        assert_eq!(state.hosting.as_ref().unwrap().members.len(), 2);
476
477        // Remove member
478        state.remove_member("node_b");
479        assert_eq!(state.hosting.as_ref().unwrap().members.len(), 1);
480
481        // End session
482        let sid = state.end_session();
483        assert!(sid.is_some());
484        assert!(state.hosting.is_none());
485
486        // Follow
487        state.follow("sess1".into(), "node_x".into(), "Xavier".into());
488        assert!(state.following.is_some());
489        assert_eq!(state.following.as_ref().unwrap().host_name, "Xavier");
490
491        // Unfollow
492        let sid = state.unfollow();
493        assert_eq!(sid.unwrap(), "sess1");
494        assert!(state.following.is_none());
495    }
