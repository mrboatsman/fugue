6/// Wrap a subsonic response body and serialize to the client's requested format.
7pub struct SubsonicResponse {
8    pub format: ResponseFormat,
9    pub body: Value,
10}
11
12impl SubsonicResponse {
13    /// Create a successful response with the given body fields merged into the
14    /// standard subsonic-response wrapper.
15    pub fn ok(format: ResponseFormat, body: Value) -> Self {
16        Self { format, body }
17    }
18
19    /// Create an empty successful response (e.g., for ping).
20    pub fn empty(format: ResponseFormat) -> Self {
21        Self {
22            format,
23            body: json!({}),
24        }
25    }
26}

28impl IntoResponse for SubsonicResponse {
29    fn into_response(self) -> Response {
30        match self.format {
31            ResponseFormat::Json | ResponseFormat::Jsonp(_) => {
32                let mut wrapper = json!({
33                    "subsonic-response": {
34                        "status": "ok",
35                        "version": "1.16.1",
36                        "type": "fugue",
37                        "serverVersion": "0.1.0",
38                        "openSubsonic": true
39                    }
40                });
41
42                // Merge body fields into subsonic-response
43                if let Some(body_obj) = self.body.as_object() {
44                    if let Some(inner) = wrapper.get_mut("subsonic-response") {
45                        if let Some(inner_obj) = inner.as_object_mut() {
46                            for (k, v) in body_obj {
47                                inner_obj.insert(k.clone(), v.clone());
48                            }
49                        }
50                    }
51                }
52
53                let json_str = serde_json::to_string(&wrapper).unwrap_or_default();
54
55                match self.format {
56                    ResponseFormat::Jsonp(ref callback) => (
57                        [("content-type", "application/javascript; charset=utf-8")],
58                        format!("{callback}({json_str})"),
59                    )
60                        .into_response(),
61                    _ => (
62                        [("content-type", "application/json; charset=utf-8")],
63                        json_str,
64                    )
65                        .into_response(),
66                }
67            }
68            ResponseFormat::Xml => {
69                let xml = build_xml_response(&self.body);
70                (
71                    [("content-type", "text/xml; charset=utf-8")],
72                    xml,
73                )
74                    .into_response()
75            }
76        }
77    }
78}

80fn build_xml_response(body: &Value) -> String {
81    let mut xml = String::from(r#"<?xml version="1.0" encoding="UTF-8"?>"#);
82    xml.push('\n');
83    xml.push_str(r#"<subsonic-response xmlns="http://subsonic.org/restapi" status="ok" version="1.16.1" type="fugue" serverVersion="0.1.0" openSubsonic="true">"#);
84
85    if let Some(obj) = body.as_object() {
86        for (key, value) in obj {
87            json_to_xml(&mut xml, key, value, 1);
88        }
89    }
90
91    xml.push_str("</subsonic-response>");
92    xml
93}

95fn json_to_xml(xml: &mut String, tag: &str, value: &Value, depth: usize) {
96    let indent = "  ".repeat(depth);
97    match value {
98        Value::Array(arr) => {
99            for item in arr {
100                json_to_xml(xml, tag, item, depth);
101            }
102        }
103        Value::Object(obj) => {
104            xml.push('\n');
105            xml.push_str(&indent);
106            xml.push('<');
107            xml.push_str(tag);
108
109            // Separate attributes (primitives) from child elements (objects/arrays)
110            let mut children = Vec::new();
111            for (k, v) in obj {
112                match v {
113                    Value::Object(_) | Value::Array(_) => {
114                        children.push((k, v));
115                    }
116                    _ => {
117                        xml.push(' ');
118                        xml.push_str(k);
119                        xml.push_str("=\"");
120                        xml.push_str(&xml_escape(&primitive_to_string(v)));
121                        xml.push('"');
122                    }
123                }
124            }
125
126            if children.is_empty() {
127                xml.push_str("/>");
128            } else {
129                xml.push('>');
130                for (k, v) in children {
131                    json_to_xml(xml, k, v, depth + 1);
132                }
133                xml.push('\n');
134                xml.push_str(&indent);
135                xml.push_str("</");
136                xml.push_str(tag);
137                xml.push('>');
138            }
139        }
140        _ => {
141            xml.push('\n');
142            xml.push_str(&indent);
143            xml.push('<');
144            xml.push_str(tag);
145            xml.push('>');
146            xml.push_str(&xml_escape(&primitive_to_string(value)));
147            xml.push_str("</");
148            xml.push_str(tag);
149            xml.push('>');
150        }
151    }
152}

154fn primitive_to_string(v: &Value) -> String {
155    match v {
156        Value::String(s) => s.clone(),
157        Value::Number(n) => n.to_string(),
158        Value::Bool(b) => b.to_string(),
159        Value::Null => String::new(),
160        _ => v.to_string(),
161    }
162}
163
164fn xml_escape(s: &str) -> String {
165    s.replace('&', "&amp;")
166        .replace('<', "&lt;")
167        .replace('>', "&gt;")
168        .replace('"', "&quot;")
169        .replace('\'', "&apos;")
170}
