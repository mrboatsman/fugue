1//! Subsonic API endpoint layer.
2//!
3//! This module implements the full set of Subsonic (and OpenSubsonic) REST
4//! endpoints that Fugue exposes to clients. Every endpoint is registered at
5//! both `/rest/<name>` and `/rest/<name>.view` to match client expectations.
6//!
7//! # Request Handling
8//!
9//! Subsonic clients vary in how they send requests:
10//! - Some use **GET** with query parameters, others use **POST** with a
11//!   form-urlencoded body
12//! - Some use the `.view` suffix on endpoint paths, others don't
13//!
14//! The [`params`] middleware normalizes this by merging POST body parameters
15//! into the query string, so handlers always read from query params regardless
16//! of the client's request method.
17//!
18//! # Authentication
19//!
20//! All endpoints (except `/admin/*`) are authenticated via the
21//! [`auth::AuthenticatedUser`] extractor, which supports three methods:
22//!
23//! 1. **Token + salt** (preferred) — `u`, `t`, `s` params; token = MD5(password + salt)
24//! 2. **Plaintext password** — `u`, `p` params; password may be hex-encoded
25//!    with `enc:` prefix
26//! 3. **API key** — `apiKey` param; SHA-256 hashed and matched against stored
27//!    keys in the database (OpenSubsonic `apiKeyAuthentication` extension)
28//!
29//! Fugue authenticates clients against its own user list (`[auth.users]` in
30//! config), then authenticates to each backend independently using the
31//! backend's stored credentials.
32//!
33//! # Response Formats
34//!
35//! Responses are returned in either XML or JSON based on the `f` query param
36//! (default: XML). The [`response`] module handles format negotiation and
37//! wrapping in the standard `<subsonic-response>` envelope.
38//!
39//! # Admin Endpoints
40//!
41//! A handful of internal endpoints (`/admin/sync`, `/admin/ticket`,
42//! `/admin/status`, etc.) are used by the CLI and are not authenticated.
43

63/// Register both `/rest/X` and `/rest/X.view` for each endpoint.
64/// Uses `any()` so both GET and POST work (many Subsonic clients use POST).
65macro_rules! subsonic_route {
66    ($router:expr, $path:expr, $handler:expr) => {
67        $router
68            .route(concat!("/rest/", $path), any($handler))
69            .route(concat!("/rest/", $path, ".view"), any($handler))
70    };
71}

165async fn fallback_handler(req: axum::extract::Request) -> axum::response::Response {
166    let path = req.uri().path().to_string();
167    tracing::warn!("unhandled endpoint: {}", path);
168    let body = format!(
169        r#"<subsonic-response xmlns="http://subsonic.org/restapi" status="ok" version="1.16.1" type="fugue" serverVersion="0.1.0" openSubsonic="true"></subsonic-response>"#,
170    );
171    axum::response::Response::builder()
172        .header("content-type", "text/xml; charset=utf-8")
173        .body(axum::body::Body::from(body))
174        .unwrap()
175}
