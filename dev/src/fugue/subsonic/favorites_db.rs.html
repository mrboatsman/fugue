6pub async fn star(
7    db: &SqlitePool,
8    owner: &str,
9    item_id: &str,
10    item_type: &str,
11) -> Result<(), FugueError> {
12    debug!("db star owner={} item_id={} type={}", owner, item_id, item_type);
13    sqlx::query(
14        "INSERT OR IGNORE INTO favorites (owner, item_id, item_type) VALUES (?, ?, ?)",
15    )
16    .bind(owner)
17    .bind(item_id)
18    .bind(item_type)
19    .execute(db)
20    .await?;
21    Ok(())
22}

24pub async fn unstar(
25    db: &SqlitePool,
26    owner: &str,
27    item_id: &str,
28) -> Result<(), FugueError> {
29    debug!("db unstar owner={} item_id={}", owner, item_id);
30    sqlx::query("DELETE FROM favorites WHERE owner = ? AND item_id = ?")
31        .bind(owner)
32        .bind(item_id)
33        .execute(db)
34        .await?;
35    Ok(())
36}

38pub async fn get_starred_ids(
39    db: &SqlitePool,
40    owner: &str,
41    item_type: &str,
42) -> Result<Vec<(String, String)>, FugueError> {
43    debug!("db get_starred_ids owner={} type={}", owner, item_type);
44    let rows: Vec<(String, String)> = sqlx::query_as(
45        "SELECT item_id, starred_at FROM favorites WHERE owner = ? AND item_type = ? ORDER BY starred_at DESC",
46    )
47    .bind(owner)
48    .bind(item_type)
49    .fetch_all(db)
50    .await?;
51    Ok(rows)
52}
